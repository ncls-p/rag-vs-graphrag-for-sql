CREATE TABLE "DRLZ1"."CP_CICS_TRANC_T"

		(

		 "DATE" DATE NOT NULL ,

		 "TIME" TIME NOT NULL ,

		 "SHIFT" CHAR(10) NOT NULL ,

		 "MVS_SYSTEM_ID" CHAR(4) NOT NULL ,

		 "CICS_SYSTEM_ID" CHAR(8) NOT NULL ,

		 "TRANSACTION_ID" CHAR(4) NOT NULL ,

		 "PROCESSOR_TYPE" CHAR(4) NOT NULL ,

		 "TERMINAL_ID" CHAR(4) DEFAULT NULL,

		 "LU_NAME" CHAR(8) DEFAULT NULL,

		 "OPERATOR_ID" CHAR(3) DEFAULT NULL,

		 "USER_ID" CHAR(8) DEFAULT NULL,

		 "TRANSACTION_TYPE" CHAR(4) DEFAULT NULL,

		 "WORKLOAD_SERVCLASS" CHAR(8) DEFAULT NULL,

		 "BRIDGE_TRANS_ID" FLOAT(8) DEFAULT NULL,

		 "MEASURED_SEC" FLOAT(8) DEFAULT NULL,

		 "RECORDS" FLOAT(8) DEFAULT NULL,

		 "TRAN_COUNT" FLOAT(8) DEFAULT NULL,

		 "CPU_SEC" FLOAT(8) DEFAULT NULL

		)

		IN "DRLSCPCE"

		ORGANIZE BY ROW;

CREATE UNIQUE INDEX "DRLZ1"."CP_CICS_TRANC_T" ON "DRLZ1"."CP_CICS_TRANC_T"

		( "DATE" ASC, "MVS_SYSTEM_ID" ASC, "TIME" ASC, "SHIFT" ASC, "CICS_SYSTEM_ID" ASC, "TRANSACTION_ID" ASC, "PROCESSOR_TYPE" ASC) CLUSTER;

@DRLZ1 Agent restructured data generate an sql query to TFP hardware usage and charges for a serial across a date range

SELECT DATE, CPU_SERIAL_NO6, MAX_MSU_USED, PURCHASED_MSU, TEMPORARY_MSU,
MSU_USAGE, HOURS_CHARGED, USAGE_CHARGE_H, USAGE_CHARGE_D, BOOST_IND
FROM "DRLZ1"."CP_CPU_TFP_HW_D"
WHERE CPU_SERIAL_NO6 = 'NNNNNN'
AND DATE BETWEEN DATE('YYYY-MM-DD') AND DATE('YYYY-MM-DD')
ORDER BY DATE;



@DRLZ1 Agent restructured data generate an sql query to top Db2 packages by CP CPU seconds for a given date and CICS system

SELECT DB2_PACKAGE, PKG_CP_CPU_SECP
FROM "DRLZ1"."CICS_DB2_TRANS_DV"
WHERE DATE = DATE('YYYY-MM-DD')
AND CICS_SYSTEM_ID = 'APPLID'
ORDER BY PKG_CP_CPU_SECP DESC
FETCH FIRST 20 ROWS ONLY;



@DRLZ1 Agent restructured data generate an sql query to total Web requests per user and system for a given week-start date

SELECT MVS_SYSTEM_ID, CICS_SYSTEM_ID, USER_ID, SUM(WEB_TOT_REQS) AS WEB_REQS
FROM "DRLZ1"."CICSWEB_TRAN_USR_W"
WHERE DATE = DATE('YYYY-MM-DD')
GROUP BY MVS_SYSTEM_ID, CICS_SYSTEM_ID, USER_ID
ORDER BY WEB_REQS DESC;
